\subsection{Python Example}

\begin{lstlisting}[language=Python, caption=Example Python code]
import numpy as np
import matplotlib.pyplot as plt

def process_data(data, threshold):
    """
    Process data based on threshold
    """
    result = []
    for value in data:
        if value > threshold:
            result.append(value)
    return np.array(result)

# Example usage
data = np.random.randn(100)
processed = process_data(data, 0.5)

# Load required libraries

# Process data
process_data <- function(data, threshold) {
  result <- data[data > threshold]
  return(result)
}
